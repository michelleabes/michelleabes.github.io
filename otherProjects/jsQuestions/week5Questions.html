                function multiples(){
                var str = ""
                for (var i = 1; i <= 100; i += 1) {
                  if (i % 3 === 0 && i % 5 === 0) {
                    //if i is a multiple of 3 and 5
                    str = str + "FooBar" + "";
                  } else if (i % 3 === 0) {
                    //if i is a multiple of 3
                    str = str + "Foo" + "";
                  } else if (i % 5 === 0) {
                    //if i is a multiple of 5
                    str = str + "Bar" + "";
                  } else {
                    //if none of the abve conditions are met, print out the number
                    str = str + i + ""
                  }
                  $("#q2result").html(str);
                }
              }

                $("#q2button").on("click", function() {
                  multiples();
                });

                function countBs(bString) {
                        var count = 0;
                        for (var i = 0; i < bString.length; i++) {
                          if (bString.charAt(i) == 'a') {
                            count = (count + 1);
                          }
                        }
                        return count;
                      }

                      });
                
                function countChar(cString, charMatch) {
                  var count = 0;
                  for (var i = 0; i < cString.length; i++) {
                    if (cString.charAt(i) == charMatch) {
                      count = (count + 1);
                    }
                  }
                  return count;
                }

                function range(start, end) {
                  var arr = [];
                  if(start <= end){

                    //this is where we establish that the numbers in the range will increase by 1
                    //and start at the start number (2) and end at the end number (6)

                  for( start; start <= end; start++){
                    arr.push(start);
                  }
                }
                  return arr;
                }

                  //2,3,4,5,6

                });

              
                var sum = function(arr) {
                  var sumVar = 0;

                  //here the code says that within our range we increase
                  //by 1 and add each number together until the range end number

                  for (var i = 0; i < arr.length; i++){

                    //the sum is sumVar (which we established as 0) plus each number in
                    //the array

                    sumVar = sumVar + arr[i];
                }
                  return sumVar;
                }

                  //here we make sumVar print to the html page

                  $("#q7result").html("The sum of the numbers is " + sumVar);
                });
              
                function reverseArray(array) {
                  var output = [];
                  for (var i = array.length - 1; i >= 0; i--){
                    output.push(array[i]);
                }
                return output;
              }

                $("#q8button").on("click", function() {



                  $("#q8result").html(reverseArray(["A", "B", "C", "D","E"]));
          });
              
                function reverseArrayInPlace(array) {
                for (var i = 0; i < Math.floor(array.length / 2); i++) {
                var old = array[i];
                array[i] = array[array.length - 1 - i];
                array[array.length - 1 - i] = old;
                }
                return array;
                }
